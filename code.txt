<script>
/* ================= CONFIG ================= */
const SHEET_ENDPOINT="https://script.google.com/macros/s/AKfycbxKVOYtyszNwsdNYX8iPZtYyYqDkxryKijLTNH73NqvjPWepLygI4UZ5jZYylBPytPF0A/exec";
const STORAGE_KEY="plant_shop_cart_v6";

/* ðŸ”µ IMPORTANT â€” change per page EXACTLY as sheet category */
const PAGE_CATEGORY="houseplants"; // spaces allowed

/* ================= STATE ================= */
let RAW=[];
let GROUPED=[];     // grouped by "group"
let cart=JSON.parse(localStorage.getItem(STORAGE_KEY)||"{}");
let hideSold=false;
let activeFilter="all";

/* ================= FETCH ================= */
async function fetchProducts(){
  const res=await fetch(`${SHEET_ENDPOINT}?ts=${Date.now()}`);
  RAW=await res.json();

  RAW.forEach(p=>{
    p.stock=Number(p.stock);
    p.price=Number(p.price);
    p.soldOut=p.stock<=0;
    p.comingSoon=(p.categories||[])
      .map(c=>c.toLowerCase())
      .includes("coming soon");
  });

  buildGroups();
  renderFilters();
  renderProducts();
  document.getElementById("loadingNotice")?.remove();
  productGrid.style.display="grid";
}

/* ================= GROUP VARIANTS ================= */
function buildGroups(){
  const pageTag=PAGE_CATEGORY.toLowerCase().trim();

  const filtered=RAW.filter(p =>
    (p.categories||[])
      .map(c=>c.toLowerCase().trim())
      .includes(pageTag)
  );

  const map={};

  filtered.forEach(p=>{
    if(!map[p.group]){
      map[p.group]={
        base:p,
        variants:[],
        categories:new Set()
      };
    }

    map[p.group].variants.push(p);
    p.categories.forEach(c=>{
      if(c.toLowerCase()!==pageTag)
        map[p.group].categories.add(c);
    });
  });

  GROUPED=Object.values(map);
}

/* ================= FILTERS ================= */
function renderFilters(){
  const row=document.getElementById("filterRow");

  const cats=new Set();

  GROUPED.forEach(g=>{
    g.categories.forEach(c=>{
      const usable = GROUPED.some(prod=>{
        if(!prod.categories.has(c)) return false;

        if(activeFilter==="all")
          return prod.variants.some(v=>!v.soldOut);

        return true;
      });
      if(usable) cats.add(c);
    });
  });

  row.innerHTML=`
<button class="filter-btn ${activeFilter==='all'?'active':''}"
 onclick="applyFilter('all')" data-filter="all">all</button>
`+
[...cats].sort().map(c=>
`<button class="filter-btn"
 data-filter="${c}"
 onclick="applyFilter('${c.replace(/'/g,"")}')">${c}</button>`
).join("")
+
`<button class="filter-btn"
 onclick="toggleSold()" data-filter="sold">hide sold out</button>`;
}

function toggleSold(){
  hideSold=!hideSold;
  renderProducts();
}

/* ================= PRODUCTS ================= */
function renderProducts(){
  const grid=document.getElementById("productGrid");
  grid.innerHTML="";
console.log("SHOP VERSION 7 LOADED");

  GROUPED.forEach(group=>{
    if(activeFilter!=="all"){
      if(!group.categories.has(activeFilter)) return;
    }

    let variants=group.variants;

    if(activeFilter==="all")
      variants=variants.filter(v=>!v.soldOut);

    if(hideSold)
      variants=variants.filter(v=>!v.soldOut);

    if(!variants.length) return;

    const selected=variants[0];
    const id=selected.id;
    const qty=cart[id]||0;

    const swatches=[...new Map(
      group.variants.map(v=>[v.swatch,v])
    ).values()];

    let addArea = qty>0
    ? `<div class="qty-row">
        <button class="qty-btn" onclick="updateQty('${id}',-1)">-</button>
        <div class="qty-num">${qty}</div>
        <button class="qty-btn" onclick="updateQty('${id}',1)">+</button>
      </div>`
    : `<button class="add-btn"
        onclick="firstAdd('${id}')"
        ${(selected.soldOut||selected.comingSoon)?'disabled':''}>
        Add
      </button>`;

    grid.innerHTML+=`
<div class="product-card
${selected.soldOut?'sold-out':''}
${selected.comingSoon?'coming-soon':''}">

<div class="product-image-wrap">
<img class="product-image" src="${selected.photo}">
${selected.soldOut?overlay("SOLD OUT","#c33"):""}
${selected.comingSoon?overlay("COMING SOON","#777"):""}
</div>

<div>
<div class="product-title">${selected.name}</div>
<div class="product-latin">${selected.latin||""}</div>
<div class="product-price">$${selected.price.toFixed(2)}</div>

<div class="swatches">
${group.variants
  .filter(v => v.swatch && v.swatch.trim() !== "")
  .map(v => {
    let color = v.swatch.trim();
    // Fix hex without #
    if(!color.startsWith("#") && !color.startsWith("rgb") && !color.startsWith("hsl")){
      color = "#" + color;
    }
    return `
      <div class="swatch ${v.id === selected.id ? 'selected' : ''}"
           style="background-color:${color}"
           onclick="selectVariant('${group.base.group}','${v.id}')">
      </div>`;
  }).join("")}
</div>
</div>

${addArea}
</div>
</div>`;
  });

  updateBadges();
  paintSwatches();
}

function overlay(txt,color){
  return `<div class="sold-out-overlay">
  <span style="background:${color}">${txt}</span></div>`;
}

/* ================= SWATCH SELECT ================= */
function selectVariant(group,id){
  const g=GROUPED.find(x=>x.base.group===group);
  const idx=g.variants.findIndex(v=>v.id===id);
  if(idx>-1){
    g.variants.unshift(g.variants.splice(idx,1)[0]);
    renderProducts();
  }
}

/* ================= FILTER APPLY ================= */
function applyFilter(f){
  activeFilter=f;
  document.querySelectorAll('.filter-btn')
    .forEach(b=>b.classList.toggle('active',b.dataset.filter===f));
  renderProducts();
}

/* ================= CART ================= */
function getProductById(id){
  return RAW.find(p=>p.id===id);
}

function saveCart(){
  localStorage.setItem(STORAGE_KEY,JSON.stringify(cart));
  updateBadges();
}

function firstAdd(id){
  const p=getProductById(id);
  if(!p||p.soldOut||p.comingSoon) return;
  cart[id]=1;
  saveCart();
  renderProducts();
}

function updateQty(id,c){
  if(!cart[id]) return;
  cart[id]+=c;
  if(cart[id]<=0) delete cart[id];
  saveCart();
  renderProducts();
}

function updateBadges(){
  const total=Object.values(cart).reduce((s,q)=>s+q,0);
  floatBadge.textContent=total;
  cartCount.textContent=total;
}

function paintSwatches(){
  document.querySelectorAll(".swatch").forEach(s => {
    const color = s.dataset.color;
    if(color){
      // Handle missing # for hex colors (sometimes appsheets/Sheets gives CE2029 instead of #CE2029)
      s.style.backgroundColor = color.startsWith("#") || color.startsWith("rgb") ? color : `#${color}`;
    }
  });
}

/* ================= INIT ================= */
fetchProducts();
setInterval(fetchProducts,300000);
</script>
